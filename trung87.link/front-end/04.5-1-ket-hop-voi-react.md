# Ứng dụng 1: Kết hợp với React (TSX)

```tsx
import React, { useState } from "react";

// 1. Định nghĩa kiểu dữ liệu cho Props
interface ProductCardProps {
  id: number;
  name: string;
  price: number;
  isAvailable: boolean;
  onAddToCart: (id: number) => void;
}

// 2. Định nghĩa kiểu dữ liệu cho State
interface CartState {
  items: number[];
  total: number;
}

const ProductCard: React.FC<ProductCardProps> = ({
  id,
  name,
  price,
  isAvailable,
  onAddToCart,
}) => {
  // Sử dụng Generic để định nghĩa kiểu cho useState
  const [cart, setCart] = useState<CartState>({ items: [], total: 0 });

  const handleAdd = () => {
    setCart({
      items: [...cart.items, id],
      total: cart.total + price,
    });
    onAddToCart(id);
  };

  return (
    <div className="border p-4 rounded-lg shadow-md">
      <h2 className="text-xl font-bold">{name}</h2>
      <p className="text-gray-600">Giá: {price.toLocaleString()}đ</p>

      {isAvailable ? (
        <button
          onClick={handleAdd}
          className="mt-3 bg-blue-500 text-white px-4 py-2 rounded"
        >
          Thêm vào giỏ ({cart.items.length})
        </button>
      ) : (
        <p className="text-red-500 mt-3 italic">Hết hàng</p>
      )}
    </div>
  );
};

export default ProductCard;
```
