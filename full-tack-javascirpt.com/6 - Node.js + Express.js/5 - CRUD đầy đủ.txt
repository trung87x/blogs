  if (!user) return res.status(404).json({ error: 'KhÃ´ng tÃ¬m tháº¥y user' });

  const { name, email } = req.body;
  if (!name || !email) return res.status(400).json({ error: 'Thiáº¿u name hoáº·c email' });

  const existed = await User.findOne({ email });
  if (existed) return res.status(409).json({ error: 'Email Ä‘Ã£ tá»“n táº¡i' });
